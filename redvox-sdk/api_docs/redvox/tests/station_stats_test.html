from datetime import timedelta
import time
from redvox.common import api_reader
from redvox.common import date_time_utils as dtu
from redvox.common.io import ReadFilter
import pandas as pd
import numpy as np


def main():
    """
    base_dir: the directory where your test dataset(s) are stored
    tests: list of dictionaries that consists of:
        :'input_dir': the specific directory within base_dir where your api900/api1000 are stored
        :'start_dt': the start datetime of your request
        :'end_dt': the end datetime of your request
    """

    test_number = 8
    base_dir = '/Users/yusukehatanaka/Desktop/DATA/'
    tests = [{'input_dir': base_dir + '20210423_80Hz_baseline/',
              'start_dt': dtu.datetime_from_epoch_microseconds_utc(1619135928951000),
              'end_dt': dtu.datetime_from_epoch_seconds_utc(1619137200)},

             # edge case: requesting all files
             {'input_dir': base_dir + '20210423_80Hz_baseline/',
              'start_dt': dtu.datetime_from_epoch_seconds_utc(1619136000),
              'end_dt': dtu.datetime_from_epoch_seconds_utc(1619137500)},

             {'input_dir': base_dir + '20210211_BigBird/',
              'start_dt': dtu.datetime_from_epoch_seconds_utc(1612987140),
              'end_dt': dtu.datetime_from_epoch_seconds_utc(1612988640)},

             {'input_dir': base_dir + '20201006_SPX_F9/',
              'start_dt': dtu.datetime_from_epoch_seconds_utc(1601983900),
              'end_dt': dtu.datetime_from_epoch_seconds_utc(1601984374)},

             {'input_dir': base_dir + '20200222_GoodComms/',
              'start_dt': dtu.datetime_from_epoch_seconds_utc(1582396300),
              'end_dt': dtu.datetime_from_epoch_seconds_utc(1582396750)},

             {'input_dir': base_dir + '20210115_Big_Island_Earthquake/',
              'start_dt': dtu.datetime_from_epoch_seconds_utc(1610684000),
              'end_dt': dtu.datetime_from_epoch_seconds_utc(1610684150)},

             # edge case: requesting all files
             {'input_dir': base_dir + '20180824_Hurricane_Lane/',
              'start_dt': dtu.datetime_from_epoch_seconds_utc(1535068800),
              'end_dt': dtu.datetime_from_epoch_seconds_utc(1535155200)},

             {'input_dir': base_dir + '20180824_Hurricane_Lane/',
              'start_dt': dtu.datetime_from_epoch_seconds_utc(1535070000),
              'end_dt': dtu.datetime_from_epoch_seconds_utc(1535140000)},

             {'input_dir': base_dir + '20201027_SDK_Example/',
              'start_dt': dtu.datetime_from_epoch_seconds_utc(1603806300),
              'end_dt': dtu.datetime_from_epoch_seconds_utc(1603808100)}]

    input_dir = tests[test_number]['input_dir']
    rf = ReadFilter(
        start_dt=tests[test_number]['start_dt'],
        end_dt=tests[test_number]['end_dt'],
        start_dt_buf=timedelta(minutes=0),
        end_dt_buf=timedelta(minutes=0)
    )
    execution_times = {'new': [], 'original': []}
    # run 10 times of both new and original algorithms to compute the execution times
    for _ in range(10):
        for run_version in ['new', 'original']:  # ['original', 'new']:
            # print(_, run_version)
            print(f'Running the {run_version} version: {_+1}/10')
            start_time = time.time()
            reader = api_reader.ApiReader(
                base_dir=input_dir,
                structured_dir=True,
                read_filter=rf,
                run_version=run_version
            )
            end_time = time.time()
            execution_times[run_version].append(end_time - start_time)
            # print('results from running version: ' + run_version)
            # print('Station ids:', reader.index_summary.station_ids())
            # print('Total packets:', reader.index_summary.total_packets())
            # print('Time passed: {:.2f}'.format(end_time - start_time))

            if _ == 0:
                holder = []
                for i in reader.files_index.entries:
                    holder.append(i.full_path)
                # print('There are ' + str(pd.unique(holder).size) + ' unique packages with different paths')
                pd.DataFrame(holder).to_csv('index_information_' + run_version + '.csv')
                print('results from running version: ' + run_version)
                print('Station ids:', reader.index_summary.station_ids())
                print('Total packets:', reader.index_summary.total_packets())
        if _ == 0:
            df_original = pd.read_csv('index_information_original.csv', names=['index', 'path'], skiprows=[0])
            df_new = pd.read_csv('index_information_new.csv', names=['index', 'path'])
            paths_original = set(df_original['path'])
            paths_new = set(df_new['path'])
    # Make sure the newer version does not miss any package
    assert len(paths_original-paths_new) == 0

    print('execution times by the new version')
    print(execution_times['new'])
    print('\nexecution times by the original version')
    print(execution_times['original'])
    print()
    print('mean and std of the new version:{:.2f}土{:.2f} sec'.format(np.mean(execution_times['new']),
                                                                     np.std(execution_times['new'])))
    print('mean and std of the original version:{:.2f}土{:.2f} sec'.format(np.mean(execution_times['original']),
                                                                          np.std(execution_times['original'])))

if __name__ == '__main__':
    main()
